# Boilerhouse Go Port — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port Boilerhouse to Go with a fully K8s-native architecture — no SQLite, no blob store, no Runtime interface. The K8s API server is the only state store.

**Architecture:** Operator controllers manage Pods/PVCs/Services directly via client-go. API server is a thin REST-to-K8s translation layer. PVCs replace the tar/extract/inject overlay cycle. Idle monitoring uses reconciler requeueAfter.

**Tech Stack:** Go 1.23+, controller-runtime v0.19, chi v5, envtest, sigs.k8s.io/yaml, testify

**Spec:** `docs/superpowers/specs/2026-04-15-go-port-design.md`

**Branch:** `experimental/go`

package operator

import (
	"testing"

	"github.com/stretchr/testify/assert"
	"github.com/stretchr/testify/require"

	v1alpha1 "github.com/zdavison/boilerhouse/go/api/v1alpha1"
)

func TestTranslate_MinimalWorkload(t *testing.T) {
	spec := v1alpha1.BoilerhouseWorkloadSpec{
		Version: "1",
		Image:   v1alpha1.WorkloadImageSpec{Ref: "alpine:latest"},
		Resources: v1alpha1.WorkloadResourcesSpec{
			VCPUs: 1, MemoryMb: 128, DiskGb: 1,
		},
		Network: &v1alpha1.WorkloadNetworkSpec{Access: "none"},
	}

	result := Translate(spec, TranslateOpts{
		InstanceId:  "inst-001",
		WorkloadName: "minimal",
		Namespace:    "boilerhouse",
	})

	// Pod assertions
	require.NotNil(t, result.Pod)
	assert.Equal(t, "inst-001", result.Pod.Name)
	assert.Equal(t, "boilerhouse", result.Pod.Namespace)
	assert.Equal(t, "alpine:latest", result.Pod.Spec.Containers[0].Image)
	assert.Equal(t, "true", result.Pod.Labels["boilerhouse.dev/managed"])
	assert.Equal(t, "minimal", result.Pod.Labels["boilerhouse.dev/workload"])

	// Security context — all caps dropped, no privilege escalation
	sc := result.Pod.Spec.Containers[0].SecurityContext
	require.NotNil(t, sc)
	assert.True(t, sc.Capabilities != nil && len(sc.Capabilities.Drop) > 0)
	assert.False(t, *sc.AllowPrivilegeEscalation)

	// NetworkPolicy — none access means deny all egress
	require.NotNil(t, result.NetworkPolicy)
	assert.Empty(t, result.NetworkPolicy.Spec.Egress)

	// No Service for workloads with no exposed ports
	assert.Nil(t, result.Service)

	// No PVC for workloads with no overlay dirs
	assert.Nil(t, result.PVC)
}

func TestTranslate_WorkloadWithOverlayDirs(t *testing.T) {
	spec := v1alpha1.BoilerhouseWorkloadSpec{
		Version:   "1",
		Image:     v1alpha1.WorkloadImageSpec{Ref: "alpine:latest"},
		Resources: v1alpha1.WorkloadResourcesSpec{VCPUs: 1, MemoryMb: 128, DiskGb: 1},
		Network:   &v1alpha1.WorkloadNetworkSpec{Access: "none"},
		Filesystem: &v1alpha1.WorkloadFilesystemSpec{
			OverlayDirs: []string{"/workspace", "/data"},
		},
	}

	result := Translate(spec, TranslateOpts{
		InstanceId:   "inst-002",
		WorkloadName: "my-agent",
		TenantId:     "alice",
		Namespace:    "boilerhouse",
	})

	// PVC created for overlay dirs
	require.NotNil(t, result.PVC)
	assert.Equal(t, "overlay-alice-my-agent", result.PVC.Name)

	// Pod mounts the PVC at each overlay dir
	var mountPaths []string
	for _, vm := range result.Pod.Spec.Containers[0].VolumeMounts {
		if vm.Name == "overlay" {
			mountPaths = append(mountPaths, vm.MountPath)
		}
	}
	// The PVC is mounted once; overlay dirs use subpaths
	assert.NotEmpty(t, result.Pod.Spec.Containers[0].VolumeMounts)
}

func TestTranslate_HealthProbe(t *testing.T) {
	spec := v1alpha1.BoilerhouseWorkloadSpec{
		Version:   "1",
		Image:     v1alpha1.WorkloadImageSpec{Ref: "alpine:latest"},
		Resources: v1alpha1.WorkloadResourcesSpec{VCPUs: 1, MemoryMb: 128, DiskGb: 1},
		Network:   &v1alpha1.WorkloadNetworkSpec{Access: "none"},
		Health: &v1alpha1.WorkloadHealthSpec{
			IntervalSeconds:    5,
			UnhealthyThreshold: 3,
			HTTPGet:            &v1alpha1.WorkloadHTTPGetSpec{Path: "/health", Port: 8080},
		},
	}

	result := Translate(spec, TranslateOpts{
		InstanceId:   "inst-005",
		WorkloadName: "agent",
		Namespace:    "boilerhouse",
	})

	probe := result.Pod.Spec.Containers[0].ReadinessProbe
	require.NotNil(t, probe)

```go
package operator

import (
	"context"

	"sigs.k8s.io/controller-runtime/pkg/client"
	"sigs.k8s.io/controller-runtime/pkg/controller/controllerutil"
	"sigs.k8s.io/controller-runtime/pkg/reconcile"
	ctrl "sigs.k8s.io/controller-runtime"

	v1alpha1 "github.com/zdavison/boilerhouse/go/api/v1alpha1"
)

const finalizerName = "boilerhouse.dev/cleanup"

type WorkloadReconciler struct {
	client.Client
}

func (r *WorkloadReconciler) Reconcile(ctx context.Context, req reconcile.Request) (reconcile.Result, error) {
	var wl v1alpha1.BoilerhouseWorkload
	if err := r.Get(ctx, req.NamespacedName, &wl); err != nil {
		return reconcile.Result{}, client.IgnoreNotFound(err)
	}

	// Handle deletion
	if !wl.DeletionTimestamp.IsZero() {
		// Cleanup: delete any Pods, PVCs, etc. for this workload
		controllerutil.RemoveFinalizer(&wl, finalizerName)
		return reconcile.Result{}, r.Update(ctx, &wl)
	}

	// Add finalizer
	if !controllerutil.ContainsFinalizer(&wl, finalizerName) {
		controllerutil.AddFinalizer(&wl, finalizerName)
		if err := r.Update(ctx, &wl); err != nil {
			return reconcile.Result{}, err
		}
	}

	// Validate spec
	if err := validateWorkloadSpec(wl.Spec); err != nil {
		wl.Status.Phase = "Error"
		wl.Status.Detail = err.Error()
		wl.Status.ObservedGeneration = wl.Generation
		return reconcile.Result{}, r.Status().Update(ctx, &wl)
	}

	// Mark ready
	wl.Status.Phase = "Ready"
	wl.Status.Detail = ""
	wl.Status.ObservedGeneration = wl.Generation
	return reconcile.Result{}, r.Status().Update(ctx, &wl)
}

func (r *WorkloadReconciler) SetupWithManager(mgr ctrl.Manager) error {
	return ctrl.NewControllerManagedBy(mgr).
		For(&v1alpha1.BoilerhouseWorkload{}).
		Complete(r)
}

func validateWorkloadSpec(spec v1alpha1.BoilerhouseWorkloadSpec) error {
	if spec.Image.Ref == "" && spec.Image.Dockerfile == "" {
		return &validationError{"either image.ref or image.dockerfile is required"}
	}
	if spec.Resources.VCPUs <= 0 {
		return &validationError{"resources.vcpus must be > 0"}
	}
	if spec.Resources.MemoryMb <= 0 {
		return &validationError{"resources.memoryMb must be > 0"}
	}
	return nil
}

type validationError struct{ msg string }
func (e *validationError) Error() string { return e.msg }
```

### Task 4.3: ClaimController

**Files:**
- Create: `go/internal/operator/claim_controller.go`
- Create: `go/internal/operator/claim_controller_test.go`

> **For implementor:** This is the most important controller. The ClaimController reconciles BoilerhouseClaim CRDs. Behavioral requirements:
>
> **Claim flow (new Claim, no deletion timestamp):**
> 1. Look up the referenced BoilerhouseWorkload — error if not found or not Ready
> 2. Check if a Pod already exists with label `boilerhouse.dev/tenant=<tenantId>`, `boilerhouse.dev/workload=<name>` and is Running → return existing (source: `existing`)
> 3. Check for a Pool Pod with `pool-status=ready` for this workload
>    - If found: relabel with `tenant=<id>`, `pool-status=acquired`. If tenant has a PVC, attach it (this requires pod recreation). Update Claim status. (source: `pool` or `pool+data`)
>    - If not found: create a new Pod using the Translator, with PVC if tenant has one. (source: `cold` or `cold+data`)
> 4. Wait for Pod Running + Ready
> 5. Update Claim status: phase=Active, instanceId, endpoint, source, claimedAt
> 6. Start idle monitoring: `requeueAfter(idle.timeout_seconds / 2)`
>
> **Idle check (reconcile of Active claim):**
> 1. Read `boilerhouse.dev/last-activity` annotation on the Claim
> 2. Look up workload idle config
> 3. If `now - lastActivity > idle.timeoutSeconds`:
>    - If action=hibernate: delete Pod (PVC persists), set Claim phase=Released
>    - If action=destroy: delete Pod AND PVC, set Claim phase=Released
> 4. If not idle: `requeueAfter(remainingTime)`
>
> **Release flow (deletion timestamp set):**
> 1. Delete the tenant's Pod for this workload
> 2. If idle action is `destroy`: also delete the PVC
> 3. Remove finalizer
>
> **PVC naming:** `overlay-<tenantId>-<workloadName>`
>
> **Attaching PVC to pool Pod:** A pool Pod has no PVC mounted. When a tenant with overlay data claims a pool Pod, the controller must delete the pool Pod and create a new Pod with the PVC mounted. The source is `pool+data`. If the tenant has no overlay data, the pool Pod is used as-is (source: `pool`).
>
> Test with envtest: test each flow (claim from pool, cold boot, existing, idle timeout, release).
